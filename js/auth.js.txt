// التحديث: تصحيح مسارات التوجيه وربط ترويسات x-device-id

const AUTH = {
    // 1. فحص الجلسة عند الدخول لأي صفحة
    checkSession: () => {
        const token = localStorage.getItem(CONFIG.TOKEN_KEY || 'moakkil_token');
        const user = localStorage.getItem(CONFIG.USER_KEY || 'moakkil_user');
        
        if (!token || !user) {
            window.location.replace('login.html');
            return null;
        }
        return JSON.parse(user);
    },

    // 2. طلب رمز التحقق (OTP)
    requestOTP: async (phone) => {
        try {
            const response = await fetch(`${CONFIG.API_URL}/api/auth/request-otp`, { [cite: 11, 14]
                method: 'POST',
                headers: { 
                    'Content-Type': 'application/json',
                    'x-device-id': localStorage.getItem('moakkil_device_id') || 'unknown'
                },
                body: JSON.stringify({ phone })
            });

            const data = await response.json(); [cite: 14]
            if (!response.ok) throw new Error(data.error || 'هذا الرقم غير مسجل أو الحساب موقوف');
            return data;
        } catch (error) {
            throw error;
        }
    },

    // 3. التحقق من الرمز وتحديد الصفحة التالية
    verifyOTP: async (phone, otp) => {
        try {
            const deviceId = localStorage.getItem('moakkil_device_id') || 'unknown';
            
            const response = await fetch(`${CONFIG.API_URL}/api/auth/verify-otp`, { [cite: 11, 14]
                method: 'POST',
                headers: { 
                    'Content-Type': 'application/json',
                    'x-device-id': deviceId 
                },
                body: JSON.stringify({ phone, otp })
            });

            const data = await response.json(); [cite: 14]
            if (!response.ok) throw new Error(data.error || 'الرمز غير صحيح');
            
            // حفظ البيانات محلياً
            localStorage.setItem(CONFIG.TOKEN_KEY || 'moakkil_token', data.token); [cite: 11, 15]
            localStorage.setItem(CONFIG.USER_KEY || 'moakkil_user', JSON.stringify(data.user)); [cite: 11, 15]
            if (data.user.firm_id) localStorage.setItem(CONFIG.FIRM_KEY, data.user.firm_id); [cite: 11, 15]

            // التوجيه الذكي (حل مشكلة الـ register)
            const userRole = data.user.role || data.user.user_type;
            
            setTimeout(() => {
                if (userRole === 'super_admin' || userRole === 'superadmin' || data.user.is_setup_required) {
                    window.location.replace('register.html'); 
                } else {
                    window.location.replace('app.html');
                }
            }, 800);

            return data;
        } catch (error) {
            throw error;
        }
    },

    // 4. الدخول بالبصمة
    loginWithBiometric: async () => {
        const savedCredId = localStorage.getItem('moakkil_biometric_id');
        const savedPhone = localStorage.getItem('moakkil_saved_phone');
        if (!savedCredId) throw new Error("البصمة غير مفعلة على هذا الجهاز");

        try {
            // ملاحظة: هنا نعتمد على دالة biometricLogin الموجودة في api.js
            const res = await API.biometricLogin({ [cite: 11, 15]
                credential_id: savedCredId, 
                phone: savedPhone,
                deviceId: localStorage.getItem('moakkil_device_id')
            });

            if (res.error) throw new Error(res.error);
            
            // التوجيه بعد نجاح البصمة
            const userRole = res.user.role || res.user.user_type;
            window.location.replace((userRole === 'super_admin' || userRole === 'superadmin') ? 'register.html' : 'app.html');
        } catch (e) {
            throw e;
        }
    }
};
